```python
class SphereWidget(tk.Canvas):
    """Orange JARVIS-style morphing blob. Idle: slow amber pulse.
    Speaking: faster morph + rings. Muted: frozen indigo. Click to toggle mute."""

    H       = 160
    N_PTS   = 24
    BG_RGB  = (6, 8, 16)   # matches BG = "#060810"

    def __init__(self, parent, on_mute_toggle=None, **kwargs):
        super().__init__(parent, height=self.H, bg=BG,
                         highlightthickness=0, **kwargs)
        self._speaking       = False
        self._muted          = False
        self._t              = 0.0
        self._rings          = [0.0, 0.333, 0.666]
        self._mute_flash     = 0
        self._on_mute_toggle = on_mute_toggle
        self.bind("<Button-1>", self._on_click)
        self._animate()

    # ── Public API ────────────────────────────────────────────────────────────

    def set_speaking(self, active: bool):
        self._speaking = active

    def set_muted(self, active: bool):
        if active and not self._muted:
            self._mute_flash = 30
        self._muted = active

    # ── Interaction ───────────────────────────────────────────────────────────

    def _on_click(self, _event):
        new_state = not self._muted
        self.set_muted(new_state)
        if self._on_mute_toggle:
            self._on_mute_toggle(new_state)

    # ── Animation ─────────────────────────────────────────────────────────────

    def _animate(self):
        try:
            if not self._muted:
                self._t += 2.0 if self._speaking else 1.0
            if self._speaking and not self._muted:
                for i in range(len(self._rings)):
                    self._rings[i] = (self._rings[i] + 0.012) % 1.0
            if self._mute_flash > 0:
                self._mute_flash -= 1
            self._draw()
            self.after(33, self._animate)
        except tk.TclError:
            pass

    # ── Helpers ───────────────────────────────────────────────────────────────

    def _blob_pts(self, cx, cy, base_r):
        """Return flat list of (x, y, ...) polygon points for a morphing blob."""
        pts = []
        t = self._t
        for i in range(self.N_PTS):
            a = 2 * math.pi * i / self.N_PTS
            r = (base_r
                 + base_r * 0.18 * math.sin(a * 3 + t * 0.05)
                 + base_r * 0.12 * math.sin(a * 5 - t * 0.03)
                 + base_r * 0.08 * math.cos(a * 7 + t * 0.07))
            pts.append(cx + r * math.cos(a))
            pts.append(cy + r * math.sin(a))
        return pts

    def _blend(self, orb_rgb, alpha):
        """Blend orb_rgb toward BG_RGB at the given alpha (0=BG, 1=orb)."""
        br, bg_, bb = self.BG_RGB
        r = int(br + (orb_rgb[0] - br) * alpha)
        g = int(bg_ + (orb_rgb[1] - bg_) * alpha)
        b = int(bb + (orb_rgb[2] - bb) * alpha)
        return f"#{r:02x}{g:02x}{b:02x}"

    # ── Draw ──────────────────────────────────────────────────────────────────

    def _draw(self):
        self.delete("all")
        w = self.winfo_width()
        if w < 10:
            return
        cx = w // 2
        cy = self.H // 2

        if self._muted:
            orb_rgb = (106, 0, 255)   # indigo
            base_r  = 30
        elif self._speaking:
            orb_rgb = (255, 170, 0)   # bright amber
            base_r  = 42
        else:
            orb_rgb = (255, 140, 0)   # deep orange
            base_r  = 38

        # Speaking rings (drawn behind blob)
        if self._speaking and not self._muted:
            for phase in self._rings:
                rr  = 24 + phase * 40
                col = self._blend(orb_rgb, (1.0 - phase) * 0.5)
                self.create_oval(cx - rr, cy - rr, cx + rr, cy + rr,
                                 outline=col, width=1)

        # Outer glow layer
        outer_pts = self._blob_pts(cx, cy, base_r * 1.6)
        self.create_polygon(*outer_pts,
                            fill=self._blend(orb_rgb, 0.10), outline="",
                            smooth=True)

        # Mid glow layer
        mid_pts = self._blob_pts(cx, cy, base_r * 1.3)
        self.create_polygon(*mid_pts,
                            fill=self._blend(orb_rgb, 0.20), outline="",
                            smooth=True)

        # Core blob
        core_pts = self._blob_pts(cx, cy, base_r)
        self.create_polygon(*core_pts,
                            fill=self._blend(orb_rgb, 0.12),
                            outline=self._blend(orb_rgb, 1.0),
                            width=2, smooth=True)

        # Inner highlight spot
        hl_x = cx - base_r // 3
        hl_y = cy - base_r // 3
        self.create_oval(hl_x - 4, hl_y - 4, hl_x + 4, hl_y + 4,
                         fill=self._blend(orb_rgb, 0.75), outline="")

        # State label
        label_y = cy + base_r + 16
        if self._mute_flash > 0:
            self.create_text(cx, cy, text="🔇",
                             font=("Consolas", 14),
                             fill=self._blend(orb_rgb, 1.0))
        elif self._muted:
            self.create_text(cx, label_y, text="MUTED",
                             font=("Consolas", 7),
                             fill=self._blend(orb_rgb, 1.0))
        elif self._speaking:
            self.create_text(cx, label_y, text="SPEAKING",
                             font=("Consolas", 7),
                             fill=self._blend(orb_rgb, 1.0))
        else:
            self.create_text(cx, label_y, text="STANDBY",
                             font=("Consolas", 7), fill=TEXT_DIM)
```

```python
    # ── Camera ────────────────────────────────────────────────────────────────

    def _start_camera(self):
        """Start camera capture thread. Falls back to NO SIGNAL if cv2 missing."""
        try:
            import cv2 as _cv2
        except ImportError:
            self._cam_available = False
            self._animate_no_signal()
            return

        def _capture():
            cap = None
            try:
                cap = _cv2.VideoCapture(0)
                if not cap.isOpened():
                    self._cam_available = False
                    self.after(0, self._animate_no_signal)
                    return
                self._cam_available = True
                while True:
                    ret, frame = cap.read()
                    if not ret:
                        break
                    try:
                        from PIL import Image, ImageTk
                        rgb   = _cv2.cvtColor(frame, _cv2.COLOR_BGR2RGB)
                        img   = Image.fromarray(rgb).resize((130, 90))
                        img_tk = ImageTk.PhotoImage(img)
                        self.after(0, self._show_cam_frame, img_tk)
                    except Exception:
                        pass
                    import time as _t; _t.sleep(0.033)
            except Exception:
                self._cam_available = False
                self.after(0, self._animate_no_signal)
            finally:
                if cap is not None:
                    cap.release()

        threading.Thread(target=_capture, daemon=True).start()

    def _show_cam_frame(self, img_tk):
        """Display a camera frame on the canvas (called from main thread)."""
        try:
            self._cam_img_ref = img_tk   # prevent GC
            self._cam_canvas.create_image(0, 0, anchor="nw", image=img_tk)
        except tk.TclError:
            pass

    def _animate_no_signal(self):
        """Animate a NO SIGNAL placeholder. Runs on the 33ms loop."""
        try:
            c = self._cam_canvas
            c.delete("all")
            # Dark background
            c.create_rectangle(0, 0, 130, 90, fill=self.HBG, outline="")
            # Scrolling scan-lines
            for i in range(6):
                y = (self._t * 2 + i * 16) % 90
                c.create_line(0, y, 130, y, fill="#0a1a18", width=1)
            # NO SIGNAL text
            c.create_text(65, 38, text="NO SIGNAL",
                          font=("Consolas", 9, "bold"), fill="#004840")
            # cv2 install hint if missing
            try:
                import cv2  # noqa: F401
            except ImportError:
                c.create_text(65, 52, text="(pip install opencv-python)",
                              font=("Consolas", 7), fill="#003830")
            # Blinking dot
            if self._t % 30 < 15:
                c.create_oval(120, 4, 128, 12, fill="#ff4444", outline="")
            self._t += 1
            self.after(33, self._animate_no_signal)
        except tk.TclError:
            pass
```

```python
    def _sr_append(self, line: str, tag: str = "ok"):
```
